local outputTable = {}

-- Modifiable Sections
outputTable["CrosshairType"] = 5 -- 0 = No crosshair, 1 = Plus, 2 = Circle, 3 = Dot, 4 = ZTZ99, 5 = Merkava
outputTable["MainColor"] = Color(255, 255, 255, 255) -- Main color of hud
outputTable["OutlineColor"] = Color(0, 0, 0, 255) -- Outline color of hud

outputTable["HasMultipleGuns"] = 1 -- 1 = Has multiple guns
outputTable["HasMainGun"] = 1 -- 1 = Has main gun, overwritten by HasMultipleGuns
outputTable["DetectAmmoTypes"] = 0 -- 1 = Automatically detects all ammo types on vehicle
outputTable["Elevation"] = 25 -- Elevation of main gun
outputTable["Depression"] = 10 -- Depression of main gun

outputTable["HasEngineInfo"] = 1 -- Has engine information
outputTable["ShowRadarData"] = 1 -- 1 = Shows input radar data
outputTable["ShowTargetTrack"] = 0 -- 1 = Shows input target data
outputTable["ShowLeadPosition"] = 0 -- 1 = Shows lead position

-- NOTE: Showing radar data is rather intensive

wire.adjustOutputs({"DATA"}, {"TABLE"}, {"Data given for use by the base itself."})
wire.ports.DATA = outputTable
